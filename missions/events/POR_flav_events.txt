##Information event for Sao Tome and fernandao po
province_event = {
	id = POR_flav.12
	title = "POR_flav.12.t"
	desc = {
	   trigger = { province_id = 1306 }
	   desc = "POR_flav.12.d.1"
	}
	desc = {
	   trigger = { province_id = 1097 }
	   desc = "POR_flav.12.d.2"
	}
	picture = COLONIZATION_eventPicture

	fire_only_once = yes

	trigger = {
		owned_by = POR
		OR = {
		colonysize = 300
		is_city = yes
		}
		OR = {
		province_id = 1306 #sao tome
		province_id = 1097 #fernando po
		}
	}

	mean_time_to_happen = {
		days = 90
	}

	option = {
		ai_chance = {
			factor = 95
		}
		name = "POR_flav.12.a.1"
		if = {
			limit = {
				province_id = 1306
			}
			1306 = { 
			add_colonysize = 200
			POR = { add_treasury = -10 }
			}
        }
		if = {
			limit = {
				AND = {
				province_id = 1097
				NOT = { province_id = 1306 }
				}
			}
			1097 = { 
			add_colonysize = 200
			POR = { 
				add_treasury = -10 
			}
			}
		}
		POR = { set_country_flag = por_colonized_volcanic_islands }
	}
	option = { #refuse
		name = "POR_flav.12.b"
		ai_chance = {
			factor = 5
		}
		POR = { 
			set_country_flag = por_colonized_volcanic_islands
		}
	}
		
}
